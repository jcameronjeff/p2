## Description

Links allow a user to navigate to another location via descriptive text.

## Ingredients

Links are made up of a combination of **elements** and **classes**. The styles will be made up of **Required** and **Selectable** .

### Required

<Kitchen>
  <input className='text-input' type='text' placeholder='placeholder' />
</Kitchen>

<Kitchen>
  <input className='text-input-lg' type='text' placeholder='placeholder' />
</Kitchen>

<Kitchen>
  <input className='text-input-sm' type='text' placeholder='placeholder' />
</Kitchen>

export const defaultArgs = {
  label: 'Label',
};

export const argTypes = {
  label: {
    control: {
      type: 'text',
    },
  },
};

export const renderInput = ({ label }) => {
  return (
    <div>
      {label && <label>{label}</label>}
      <input className='input' type='text' placeholder='placeholder' />
    </div>
  );
};

export const InputKitchen = (props) => (
  <Kitchen
    args={defaultArgs}
    argTypes={argTypes}
    render={renderInput}
    targetSelector='.input'
    {...props}
  />
);

<InputKitchen />
